Helper: validación centralizada
const validarLibro = ({ titulo, autor, anio }) => { const errors = []; if (!titulo || String(titulo).trim() === "") errors.push("El título no puede estar vacío."); if (!autor || String(autor).trim() === "") errors.push("El autor es obligatorio."); if (!Number.isInteger(anio) || anio <= 1500) errors.push("El año debe ser > 1500."); return { valid: errors.length === 0, errors }; };
Ruta: GET /libros con filtro
app.get("/libros", (req, res) => { const { autor } = req.query; const resultado = autor ? libros.filter(l => l.autor.toLowerCase() .includes(autor.toLowerCase())) : libros; res.status(200).json(jsonResponse( true, resultado, `${resultado.length} libro(s) encontrado(s).` )); });
